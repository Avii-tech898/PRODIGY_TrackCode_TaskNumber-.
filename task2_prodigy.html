# ============================================
# Exploratory Data Analysis (EDA)
# Titanic gender_submission.csv
# ============================================

# Import Libraries
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Set Style
sns.set(style="whitegrid")

# Load Dataset
df = pd.read_csv(r"C:\Users\avani\Downloads\gender_submission.csv")

# ============================================
# Basic Information
# ============================================

print("="*50)
print("First 5 Rows")
print(df.head())

print("\n"+"="*50)
print("Dataset Shape")
print(df.shape)

print("\n"+"="*50)
print("Column Names")
print(df.columns)

print("\n"+"="*50)
print("Data Types")
print(df.dtypes)

print("\n"+"="*50)
print("Dataset Information")
print(df.info())

print("\n"+"="*50)
print("Summary Statistics")
print(df.describe())

# ============================================
# Missing Values
# ============================================

print("\n"+"="*50)
print("Missing Values")
print(df.isnull().sum())

# ============================================
# Duplicate Values
# ============================================

print("\n"+"="*50)
print("Duplicate Rows")
print(df.duplicated().sum())

# ============================================
# Survival Count
# ============================================

print("\n"+"="*50)
print("Survival Count")
print(df["Survived"].value_counts())

# ============================================
# Survival Percentage
# ============================================

print("\n"+"="*50)
print("Survival Percentage")
print(df["Survived"].value_counts(normalize=True)*100)

# ============================================
# Bar Chart
# ============================================

plt.figure(figsize=(6,5))
sns.countplot(x="Survived", data=df)

plt.title("Survival Count")
plt.xlabel("Survived (0 = No, 1 = Yes)")
plt.ylabel("Number of Passengers")

plt.show()

# ============================================
# Pie Chart
# ============================================

plt.figure(figsize=(6,6))

df["Survived"].value_counts().plot(
    kind="pie",
    autopct="%1.1f%%",
    labels=["Not Survived","Survived"],
    startangle=90
)

plt.title("Survival Percentage")
plt.ylabel("")

plt.show()

# ============================================
# Histogram
# ============================================

plt.figure(figsize=(6,5))

plt.hist(df["PassengerId"], bins=20, edgecolor="black")

plt.title("Passenger ID Distribution")
plt.xlabel("Passenger ID")
plt.ylabel("Frequency")

plt.show()

# ============================================
# Boxplot
# ============================================

plt.figure(figsize=(6,4))

sns.boxplot(x=df["PassengerId"])

plt.title("Passenger ID Boxplot")

plt.show()

# ============================================
# Correlation Matrix
# ============================================

plt.figure(figsize=(5,4))

sns.heatmap(df.corr(), annot=True, cmap="Blues")

plt.title("Correlation Heatmap")

plt.show()

# ============================================
# Conclusion
# ============================================

print("\n"+"="*50)
print("EDA Completed Successfully!")

print("""
Key Findings:

1. Dataset contains 418 rows and 2 columns.
2. No missing values.
3. No duplicate records.
4. Around 36% passengers survived.
5. Around 64% passengers did not survive.
6. Dataset is clean and ready for analysis.
""")
