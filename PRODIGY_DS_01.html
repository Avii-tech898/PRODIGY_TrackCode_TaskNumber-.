import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset
df = pd.read_csv(r"C:\Users\avani\Downloads\API_SP.POP.TOTL_DS2_EN_csv_v2_3107\API_SP.POP.TOTL_DS2_EN_csv_v2_3107.csv", skiprows=4)

# Select Country Name and Population (2025)
population = df[['Country Name', '2025']].dropna()

# Remove regional aggregates (optional)
population = population.sort_values(by='2025', ascending=False).head(10)

# Plot
plt.figure(figsize=(10,6))
plt.bar(population['Country Name'], population['2025'])
plt.title("Top 10 Countries by Population (2025)")
plt.xlabel("Country")
plt.ylabel("Population")
plt.xticks(rotation=45, ha='right')
plt.tight_layout()
plt.show()
